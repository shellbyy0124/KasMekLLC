import json
from datetime import datetime

with open('/home/shellbyy/Desktop/BusinessSoftwareIdeas/KasMekLLC/payroll/timesheets/04.20.21.json', 'r', encoding='utf-8-sig') as f:
    data = json.load(f)

totals = []
for employee_id in data:
    start = data[employee_id]["time_in"]
    end = data[employee_id]["time_out"]
    new_start = datetime.strptime(f"{start}", "%m/%d/%y, %H:%M:%S")
    new_end = datetime.strptime(f"{end}", "%m/%d/%y, %H:%M:%S")
    data[employee_id]["total_time"] = new_end - new_start
    totals.append(str(data[employee_id]["total_time"]))

print(f"The variable start holds this data: {start}")
print(f"The variable end hold this data: {end}")
print(f"The variable new_start holds this data: {str(new_start)}")
